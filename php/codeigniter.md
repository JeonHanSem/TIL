
##url form 검증
	function valid_url($url) {
		if($url == NULL) return TRUE;
		$pattern = "|^http(s)?://[a-z0-9-]+(.[a-z0-9-]+)*(:[0-9]+)?(/.*)?$|i";
		if(!preg_match($pattern, $url)){
			$this->form_validation->set_message('유효하지 않은 url입니다');
			return FALSE;
		}
		return TRUE;
	}
